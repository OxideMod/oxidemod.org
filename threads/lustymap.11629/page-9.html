    if((currentz + r) >= 0 && (currentz + r) <= 19 && (currentx + c) >= 0 && (currentx + c) <= 19) {
    gui.url("MinimapBG", "Map", maplink, sx + " " + sy, ex + " " + ey, "0.9");

    gui.url("MinimapBG", "Map", mapcomplex + "blank.jpeg", sx + " " + sy, ex + " " + ey, "0.9");
}
