    bool ShouldBlockEscape(string target, string source, List<string> sourceMembers = null)
        {
            if (target == source)
            {
                if (raiderBlock) return true;
                return false;
            }
            if (sourceMembers is List<string> && sourceMembers.Count > 0 && sourceMembers.Contains(target))
                return false;

            return true;
        }
