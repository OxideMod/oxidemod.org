    Code:
    {
  "DatePreset": {
    "presetDay": 3,
    "presetMonth": 6,
    "presetYear": 2020,
    "setPresetDate": true   // 'true' sets on load the predefined game date
  },
  "Settings": {
    "freezeDate": true,   // 'true' would keep always the same day
  },
}
