    Code:
    {
  "MaxLimits": {
    "campfire": 25,
    "skull.fire.pit": 25,   
    "ceilinglight": 25,   
    "searchlight": 25,   
    "tunalight": 25,   
    "furnace": 25,
    "furnace.large": 25,

    "gates.external.high.stone": 25
