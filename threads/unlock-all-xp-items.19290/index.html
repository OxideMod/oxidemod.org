  8. Caffeine

    Here's code that sets the players level to 300(not sure what the max level is, you can just change the 300 to whatever level) and also unlocks everything too, although the UnlockAllItems function was written by The FSM and I haven't tested it. It does this for each player that connects, you can just save this to XPUnlocked.cs to test it out


    Code:
    using System.Collections.Generic;
using Oxide.Core;
using Oxide.Core.Configuration;
using Rust.Xp;namespace Oxide.Plugins
{
    [Info("XPUnlocked", "Caffeine", "0.1.0", ResourceId = 0)]
    class XPUnlocked : RustPlugin
    {
 
        void OnPlayerInit(BasePlayer player)
        {
            UnlockAllItems(player);
            SetLevel(player, 300);
        }
     
        /* set the players level */
        void SetLevel(BasePlayer player, int level)
         {
             ResetXP(player);
             AddXP(player,LevelToXp(level));
         }
      
         /* clear all of the players xp */
         void ResetXP(BasePlayer player)
         {
            player.xp.Reset();
         }
      
        /* give the player xp */
         void AddXP(BasePlayer player, float amount)
         {
             player.xp.Add(Rust.Xp.Definitions.Cheat, amount);
         }        /* Utility functions to return the amounts based on the input */
        float LevelToXp(int level)
        {
            return Rust.Xp.Config.LevelToXp(level);
        }
   
        float XpToLevel(float xp)
        {
            return Rust.Xp.Config.XpToLevel(xp);
        }
      
         void UnlockAllItems(BasePlayer player)
         {
            foreach(ItemDefinition i in ItemManager.itemList)
            {
                player.blueprints.Unlock(i);
            }
         }
 
    }
}

{
    [Info("XpUnlocker", "Caffeine", "0.1.0", ResourceId = 0)]
    [Description("Unlocks items without spending XP")]    class XpUnlocker : RustPlugin
    {
        void OnPlayerInit(BasePlayer player)
        {
            // Unlock all items for player
            foreach (var item in ItemManager.itemList) player.blueprints.Unlock(item);            // Set player to max level
            player.xp.Reset();
            player.xp.Add(Rust.Xp.Definitions.Cheat, Rust.Xp.Config.XpToLevel(101));
        }
    }
}

  19. Caffeine

    Here's how you can lock everything again

    Code:
    void LockAllItems(BasePlayer player)
{
            var playerInfo = ServerMgr.Instance.persistance.GetPlayerInfo(player.userID);
            playerInfo.unlockedItems.Clear();
            ServerMgr.Instance.persistance.SetPlayerInfo(player.userID, playerInfo);
            player.SendNetworkUpdate(BasePlayer.NetworkQueue.Update);
}
    I copied lot of this from PlayerBlueprints code in Assembly-CSharp.dll. The playerinfo.unlockedItems is just a regular list consisting of only item ids, so you can add, remove, clear etc...it like a regular list. By the way when you lock the items, it still shows them only they need to spend xp on them again(and be high enough level).


    If you want to list all item names and id in rcon you can do

    Code:
    void ListAllItems()
{
       foreach(ItemDefinition i in ItemManager.itemList)
        {
             Puts(string.Format("name: {0} id:{1}",i,i.itemid));
         }
  }
