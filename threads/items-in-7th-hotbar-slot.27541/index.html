  12. redBDGR

    Load this plugin & it will reset all online & sleeping players belt slots back to 6.
    Source:
    Code:
    namespace Oxide.Plugins
{
    [Info("InventoryReset", "redBDGR", "1.0.0")]
    [Description("Reset the inventory of all players")]    class InventoryReset : RustPlugin
    {
        private void Init()
        {
            foreach (BasePlayer player in BasePlayer.activePlayerList)
                player.inventory.containerBelt.capacity = 6;
            foreach (BasePlayer player in BasePlayer.sleepingPlayerList)
                player.inventory.containerBelt.capacity = 6;
        }
    }
}
