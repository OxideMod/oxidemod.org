    Code:
    // ItemAmountRanged
public override float GetAmount()
{
    if (this.maxAmount > 0f && this.maxAmount > this.amount)
    {
        return UnityEngine.Random.Range(this.amount, this.maxAmount);
    }
    return this.amount;
}
