    PLUGIN.Title = "Helptext"
    PLUGIN.Description = "Hooks into plugins to send helptext"
    PLUGIN.Author = "#Domestos"
    PLUGIN.Version = V(1, 1, 1)
    PLUGIN.HasConfig = false
    PLUGIN.ResourceID = 676

    function PLUGIN:Init()
    command.AddChatCommand("help", self.Object, "cmdHelp")
    command.AddChatCommand("version", self.Object, "cmdVersion")
    end

    function PLUGIN:cmdHelp(player)
    if not player then return end
    local args = {player}
    plugins.CallHook("SendHelpText", util.TableToArray(args))
    end

    function PLUGIN:cmdVersion(player)
    local version = tostring(Oxide.Core.OxideMod.Version)
    version = string.match(version, "(%d+.%d+.%d+)")
    player:ChatMessage("Oxide v"..version)
    end
