    CHange the config betterchat.json i think, and change the Title

                SetConfig("default", new Dictionary<string, object> {
                { "Formatting", "{Title} {Name}<color={TextColor}>:</color> {Message}" },
                { "ConsoleFormatting", "{Title} {Name}: {Message}" },
                { "Permission", "player" },
                { "Title", "[Player]" },
                { "TitleColor", "#C4FF00" },
                { "NameColor", "#DCFF66" },
                { "TextColor", "white" },
