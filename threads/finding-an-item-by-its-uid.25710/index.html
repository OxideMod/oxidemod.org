  10. Ryan

    I've figured out a little hack

    Code:
    namespace Oxide.Plugins
{
    [Info("SkullGive", "Ryan", "1.0.0")]
    [Description("Description here!")]

    class SkullGive : RustPlugin
    {
        void GiveItem(Item item, BasePlayer player, BasePlayer target)
        {
            item.name = $"Skull of \"{target.displayName}\"";
            item.text = target.UserIDString;
            player.GiveItem(item);
            PrintToChat(player, $"You've been given a Skull of \"{target.displayName}\"");
        }

        void CanMoveItem(Item item, PlayerInventory playerLoot, uint targetContainer, int targetSlot)
        {
            PrintToChat(item.GetOwnerPlayer(), item.text);
        }


        [ChatCommand("skull")]
        void skullCmd(BasePlayer player, string command, string[] args)
        {
            if(args.Length == 0 || args.Length == 1)
            {
                PrintToChat(player, "Invalid arguments. Usage: /skull give player");
                return;
            }
            if(args.Length == 2)
            {
                switch(args[0].ToLower())
                {
                    case "give":
                        var target = BasePlayer.Find(args[1]);
                        if (!target)
                        {
                            PrintToChat(player, "No player found");
                            break;
                        }
                        GiveItem(ItemManager.CreateByName("skull.human"), player, target);
                        break;
                    default:
                        PrintToChat(player, "Invalid arguments. Usage: /skull give player");
                        break;
                }
            }
        }
    }
}
    Only problem is, not sure what the item.text is used for, nothing noticeable it seems. But, you can use the item.text to check if they can do the commands you were on about.

    @Togoshige You don't necessarily need to find the Item to do what you are trying to achieve. There are other options that you could do to get the same results. For example you could simply recorded the item uid, along with the user ID of the victim, then try to match that uid to the active item when the chat command is run.

    Here is just one example of alternative options
    Code:
            Hash<uint, ulong> skullIds = new Hash<uint, ulong>(); // Save this to data       

        void GiveItem(BasePlayer targetPlayer, BasePlayer victimPlayer)
        {
            Item item = ItemManager.CreateByItemID(960793436, 1, 0);
            item.name = $"Skull of \"{victimPlayer.displayName}\"";
            skullIds[item.uid] = victimPlayer.userID;
            targetPlayer.GiveItem(item, BaseEntity.GiveItemReason.PickedUp);
        }

        [ChatCommand("example")]
        void cmdExample(BasePlayer player, string command, string[] args)
        {
            Item item = player.GetActiveItem();
            if (item != null && item.info.itemid == 960793436)
            {
                ulong targetId = 0U;
                if (skullIds.TryGetValue(item.uid, out targetId))
                {
                    // Player holds the skull of the target
                    BasePlayer targetExample1 = covalence.Players.FindPlayerById(targetId.ToString())?.Object as BasePlayer;
                    BasePlayer targetExample2 = BasePlayer.activePlayerList.First(x => x.userID == targetId) ?? BasePlayer.sleepingPlayerList.First(x => x.userID == targetId);
                }
            }
        }
