    Code:
    for (int i = 0; i < events.Length; i++)
                {
                    var prefab = events[i];
                    var name = prefab?.targetPrefab?.resourcePath ?? "Unknown";
                    if (name.Contains("heli"))
                    {
                        GameObject.Destroy(prefab);
                        Puts("Disabled default Helicopter spawning.");
                        break;
                    }
                }
