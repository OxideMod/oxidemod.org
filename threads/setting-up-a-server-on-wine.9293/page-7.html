    @Infinitypl

    For LXC Ubuntu 14.04.3 (quickly, to be improved) :
    Code:
    sudo su
apt-get update
apt-get upgrade -y
mkdir /opt/rust
STEAMCMD_URL=http://media.steampowered.com/installer/steamcmd_linux.tar.gz
echo "deb http://ppa.launchpad.net/ubuntu-wine/ppa/ubuntu trusty main" > /etc/apt/sources.list.d/wine.list
dpkg --add-architecture i386
apt-key adv --keyserver keyserver.ubuntu.com --recv-keys 5A9A06AEF9CB8DB0
apt-get update
apt-get install -y --no-install-recommends wine1.7-amd64 lib32gcc1 xvfb xauth libdbus-1-3
mkdir /opt/steamcmd
cd /opt/steamcmd/
apt-get install -y byobu wget nano
wget $STEAMCMD_URL
tar -xzf steamcmd*.tar.gz
/opt/steamcmd/steamcmd.sh +@sSteamCmdForcePlatformType windows +login anonymous +force_install_dir /opt/rust +app_update 258550 -beta experimental validate +quit
cd /opt/rust/
byobu
xvfb-run --auto-servernum --server-args='-screen 0 640x480x24:32' wine64 RustDedicated.exe -batchmode +server.hostname "SERVER NAME" +server.port 28015 +server.identity "rustserver" +server.maxplayers 250 +rcon.port 28016 +rcon.password "RCON_PASSWORD" +rcon.ip 0.0.0.0 -logFile "/opt/rust/out.txt"
    Install Oxide after...
