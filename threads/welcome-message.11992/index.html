  5. mvrb

    Code:
    namespace Oxide.Plugins
{
    [Info( "WelcomeMessage", "mvrb", "1.0.0" )]
    class WelcomeMessage : RustPlugin
    {
        void OnPlayerInit( BasePlayer player )
        {
            player.ChatMessage( "Welcome to ServerName!" );
        }
    }
}
    Save it as WelcomeMessage.cs and edit the "Welcome to ServerName!" message.
