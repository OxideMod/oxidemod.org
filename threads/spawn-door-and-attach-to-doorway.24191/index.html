    I'm not sure about attaching a door to a doorway programmatically, but locks is something I've done before:
    Code:
    Door door = SomeProcedureToGetADoor();

// code lock
CodeLock myLock = (CodeLock)GameManager.server.CreateEntity("assets/prefabs/locks/keypad/lock.code.prefab");
// or key lock
KeyLock myLock = (KeyLock)GameManager.server.CreateEntity("assets/prefabs/locks/keylock/lock.key.prefab");

myLock.SetParent(door, door.GetSlotAnchorName(BaseEntity.Slot.Lock));
myLock.OnDeployed(door);
myLock.Spawn();
door.SetSlot(BaseEntity.Slot.Lock, myLock);
