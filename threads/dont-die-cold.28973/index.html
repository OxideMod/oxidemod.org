  4. HOUGAN_Y

    The easiest way:
    Code:
    object OnEntityTakeDamage(BaseCombatEntity entity, HitInfo info)
{
     if (entity is BasePlayer && info.damageTypes.GetMajorityDamageType() == DamageType.Cold
)
          return false;
     return null;
}
