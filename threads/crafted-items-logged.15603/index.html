  2. sqroot

    Here's a plugin I wrote real quick:
    Code:
    import ConVar.Server as svclass CraftLog:
    def __init__(self):
        self.Title = "CraftLog"
        self.Description = "Logs crafting."
        self.Author = "sqroot"
        self.Version = V(1, 0, 0)
        self.ResourceId = -1    def log(self, task, msg):
        player = task.owner.displayName
        n = task.amount
        item_name = task.blueprint.targetItem.displayName.english
        sv.Log("oxide/logs/crafted.txt", msg % (player, n, item_name))    def OnItemCraft(self, task, item):
        self.log(task, "%s started: %dx %s")    def OnItemCraftCancelled(self, task):
        self.log(task, "%s cancelled at: %dx %s")
    If you want I can properly publish this as plugin as well.
