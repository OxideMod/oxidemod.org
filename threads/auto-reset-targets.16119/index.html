  9. k1lly0u

    After a quick look in assembly here is some code for the others that are making this. although it seems pointless many people making it
    Code:
    void OnEntityTakeDamage(BaseCombatEntity entity, HitInfo hitinfo)
        {          
            try
            {
                if (entity is ReactiveTarget && hitinfo.Initiator is BasePlayer)
                {                  
                    var target = (ReactiveTarget)entity;
                    if (target.IsKnockedDown())
                        timer.Once(5, ()=>
                        target.SetFlag(BaseEntity.Flags.On, true));
                }
            }
            catch (Exception ex)
            {
            }
        }
