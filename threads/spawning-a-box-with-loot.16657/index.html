  4. k1lly0u

    create the loot itself, then move it to the container

    Code:
     [ChatCommand("chest")]
        private void cmdChatairbox(BasePlayer player, string command, string[] args)
        {
            string name = "assets/prefabs/misc/supply drop/supply_drop.prefab";            Vector3 chest = new Vector3(); // You forgot this
            chest.x = UnityEngine.Random.Range(25, 27);
            chest.y = 44;
            chest.z = UnityEngine.Random.Range(568, 570);            BaseEntity entity = GameManager.server.CreateEntity(name, chest);
            entity.Spawn(true);            StorageContainer container = entity.GetComponent<StorageContainer>();
            if (container != null)
            {
                Item item = BuildItems("rifle.ak", 1); // Create a item
                if (item != null)
                    item.MoveToContainer(container.inventory); // Put in container
            }
        }
        private Item BuildItems(string shortname, int amount)
        {
            var definition = ItemManager.FindItemDefinition(shortname); // Find the item definition from its shortname
            if (definition != null)
            {
                Item item = ItemManager.CreateByItemID(definition.itemid, amount, false); // Create the item itself
                if (item != null)
                    return item;
            }         
            return null;
        }
