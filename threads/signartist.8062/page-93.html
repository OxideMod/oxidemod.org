    {

  "Error": "Image loading fail! Error: {error}",
  "ForceJPG": false,
  "JPGCompression": 85,

    Change the " "Command cooldown after url": 20.0, " to "Command cooldown after url": 0.0," or even " "Command cooldown after url": 1.0,"
