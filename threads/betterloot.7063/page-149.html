    Code:
    {
  "ItemListBarrels": {

    "ammo.handmade.shell": 50,
    "ammo.pistol": 50,
    "ammo.pistol.fire": 50

    "flameturret": 4,
    "pistol.python": 1,
    "rifle.ak": 1
     ...

  }
}
