    Code:
    function PLUGIN:OnServerInitialized()
   self:SetStackSizes();
endfunction PLUGIN:SetStackSizes()
   local tbl = {
     ["bone_fragments"] = 2000,
     ["charcoal"] = 1000,
     ["cloth"] = 2000,
     ["fat_animal"] = 2000,
     ["gunpowder"] = 2000,
     ["lowgradefuel"] = 2000,
     ["metal_fragments"] = 2000,
     ["metal_refined"] = 2000,
     ["paper"] = 2000,
     ["stones"] = 3000,
     ["sulfur"] = 2000,
     ["sulfur_ore"] = 3000,
     ["metal_ore"] = 3000,
     ["wood"] = 3000,
   }   for k,v in pairs(tbl) do
     local item = global.ItemManager.CreateByName(k,1);
     item.info.stackable = v;
   end
end
