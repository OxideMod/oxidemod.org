  9. Nathan 3

    I made edits to damage controller in the past, You could easily edit this code to do what you want.

    Code:
    private void OnEntityTakeDamage(BaseCombatEntity entity, HitInfo hitInfo)
        {
            var weapon = hitInfo?.Weapon?.GetItem()?.info?.shortname;

            if (entity == null)
                return;
            if (weapon == null || weapon == "") return;

           
            if (entity is BuildingBlock)
            {
                var block = entity as BuildingBlock;
               
                if (block == null)
                   return;
                else if (block.grade == BuildingGrade.Enum.Wood && weapon != null && weapon == "salvaged.sword")
                    hitInfo.damageTypes.Scale(Rust.DamageType.Slash, 15);
                else if (block.grade == BuildingGrade.Enum.Stone && weapon != null && weapon == "salvaged.sword")
                    hitInfo.damageTypes.Scale(Rust.DamageType.Slash, 70);
               
            }
           
                if (entity is Door && weapon == "salvaged.sword")
                hitInfo.damageTypes.Scale(Rust.DamageType.Slash, 5);
                else return;
        }
