    You do realize that the RCon banlist is to block certain IPs from connecting to the server RCon console and not the game right? When a player connects it only checks the banlist (check snippet at the bottom).

    I would also recommend if you would use your own ban system for this to check both IP and SteamId, and possible account sharing if they get really crafty. They could always change their IP and create a new family shared account at that point

    ConnectionAuth.OnNewConnection(Connection connection):
    Code:
    public void OnNewConnection(Connection connection)
{
    connection.connected = false;
    if ((connection.token == null) || (connection.token.Length < 0x20))
    {
        Reject(connection, "Invalid Token");
    }
    else if (connection.userid == 0)
    {
        Reject(connection, "Invalid SteamID");
    }
    else
    {
        if (connection.protocol != 0x515)
        {
            if (!DeveloperList.Contains(connection.userid))
            {
                Reject(connection, "Incompatible Version");
                return;
            }
            Debug.Log("Not kicking " + connection.userid + " for incompatible protocol (is a developer)");
        }
        if (ServerUsers.Is(connection.userid, ServerUsers.UserGroup.Banned))
        {
            Reject(connection, "You are banned from this server");
        }
        else
        {
            if (ServerUsers.Is(connection.userid, ServerUsers.UserGroup.Moderator))
            {
                Debug.Log(connection.username + " has auth level 1");
                connection.authLevel = 1;
            }
            if (ServerUsers.Is(connection.userid, ServerUsers.UserGroup.Owner))
            {
                Debug.Log(connection.username + " has auth level 2");
                connection.authLevel = 2;
            }
            if (DeveloperList.Contains(connection.userid))
            {
                Debug.Log(connection.username + " is a developer");
                connection.authLevel = 1;
            }
            if (this.IsConnected(connection.userid))
            {
                Reject(connection, "You are already connected!");
            }
            else
            {
                object[] args = new object[] { connection };
                if (Interface.CallHook("OnUserApprove", args) == null)
                {
                    this.m_AuthConnection.Add(connection);
                    base.StartCoroutine(this.AuthorisationRoutine(connection));
                }
            }
        }
    }
}
